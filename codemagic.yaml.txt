workflows:
  ios-pwa-build:
    name: iOS Build
    max_build_duration: 60
    environment:
      vars:
        XCODE_WORKSPACE: "ukcemap.co.uk.xcworkspace"
        XCODE_SCHEME: "ukcemap.co.uk"
    scripts:
      - name: Install CocoaPods dependencies
        script: |
          cd ukcemap.co.uk
          pod install
      - name: Build iOS app
        script: |
          xcodebuild -workspace "$XCODE_WORKSPACE" \
                     -scheme "$XCODE_SCHEME" \
                     -configuration Release \
                     -sdk iphoneos \
                     -archivePath build/UKCEMAP.xcarchive archive
      - name: Export IPA
        script: |
          xcodebuild -exportArchive \
                     -archivePath build/UKCEMAP.xcarchive \
                     -exportOptionsPlist exportOptions.plist \
                     -exportPath build/IPA
    artifacts:
      - build/IPA/*.ipa
